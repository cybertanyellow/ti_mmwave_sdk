    44    /*!
    45     *  ======== Hwi ========
    46     *  Hardware Interrupt Support Module.
    47     *
    48     *  //TODO Note: FIQ interrupt handler must save FIQ context before using FPU
    49     *
    50     *  This Hwi module provides PL192 Vectored Interrupt Controller (VIC)
    51     *  specific implementations of the APIs defined in
    52     *  {@link ti.sysbios.interfaces.IHwi IHwi}.
    53     *
    54     *  Additional ARM device-specific APIs are also provided.
    55     *
    56     *  @a(Minimal Latency Interrupts)
    57     *  For applications requiring extremely low interrupt latency, this Hwi module
    58     *  allows the user to create FIQ interrupts that have a light weight interrupt
    59     *  dispatcher. Though not a precisely correct classification, these interrupts
    60     *  are referred to as "Zero latency" interrupts.
    61     *
    62     *  "FIQ" aka "Zero latency" interrupts can be created by setting the Hwi_Param
    63     *  {@link #type} to Hwi_Type_FIQ. FIQ interrupts offer low interrupt latency
    64     *  as they do not have to pass through the regular SYS/BIOS interrupt
    65     *  dispatcher and are always enabled. When auto nesting is enabled and masking
    66     *  option ALL or LOWER is used, some or all of the VIC interrupts will
    67     *  be disabled while the interrupt is being serviced. However, none of the
    68     *  "Zero latency" interrupts are disabled (masked).
    69     *
    70     *  Unlike regular IRQ interrupts, FIQ interrupts do not run on the System stack
    71     *  but on their own FIQ stack. The stack pointer, size and section name for the
    72     *  FIQ stack can be set using the {@link #fiqStack}, {@link #fiqStackSize}
    73     *  and {@link #fiqStackSection} module wide configuration params.
    74     *
    75     *  @a(Constraints of using FIQ aka Zero latency interrupts)
    76     *  FIQ Interrupts bypass the regular SYS/BIOS interrupt dispatcher and are
    77     *  therefore not allowed to call ANY SYS/BIOS APIs that affect thread
    78     *  scheduling. Examples of API that should not be invoked are:
    79     *
    80     *  @p(dlist)
    81     *    - Swi_post(),
    82     *    - Semaphore_post(),
    83     *    - Event_post(),
    84     *    - Task_yield()
    85     *  @p
    86     *
    87     *  Here's an example showing how to create a Hwi of FIQ type:
    88     *
    89     *  @p(code)
    90     *  *.cfg:
    91     *  var Hwi = xdc.useModule('ti.sysbios.family.arm.pl192.Hwi');
    92     *  Hwi.fiqStackSize = 2048;
    93     *  Hwi.fiqStackSection = ".myFiqStack"
    94     *  Program.sectMap[".myFiqStack"] = "RAM";
    95     *
    96     *  *.c:
    97     *  #include <xdc/std.h>
    98     *  #include <xdc/runtime/System.h>
    99     *
   100     *  #include <ti/sysbios/BIOS.h>
   101     *  #include <ti/sysbios/family/arm/pl192/Hwi.h>
   102     *
   103     *  #include <xdc/cfg/global.h>
   104     *
   105     *  Void myIsrFIQ(UArg arg)
   106     *  {
   107     *      ...
   108     *  }
   109     *
   110     *  Void main(Void)
   111     *  {
   112     *      Hwi_Params hwiParams;
   113     *
   114     *      Hwi_Params_init(&hwiParams);
   115     *      hwiParams.type = Hwi_Type_FIQ;
   116     *      Hwi_create(INT_NUM_FIQ, myIsrFIQ, &hwiParams, NULL);
   117     *      ...
   118     *
   119     *      BIOS_start();
   120     *  }
   121     *  @p
   122     *
   123     *  @a(Interrupt Latency)
   124     *  This module supports daisy-chained Vectored Interrupt Controllers (VICs).
   125     *  In a daisy-chained configuration, the VIC closest to the core has the lowest
   126     *  interrupt latency. In other words, lower interrupt numbers have lower
   127     *  latency.
   128     *
   129     *  @a(Interrupt Priorities)
   130     *  FIQ interrupts have the highest priority i.e. any interrupt of FIQ type will
   131     *  have a higher priority than any other interrupt of IRQ type.
   132     *
   133     *  IRQ interrupts (or vectored interrupts in PL192 VIC terminology) can be
   134     *  assigned different priority levels, ranging from 0-15. Lower priority levels
   135     *  imply a higher effective interrupt priority. Priority level can be set by
   136     *  writing to the {@link #priority} Hwi param. By default, all interrupts are
   137     *  set to the lowest priority level.
   138     *
   139     *  If multiple interrupts that are at the same priority level and belong to
   140     *  the same VIC, occur at the time, the fixed hardware priority levels are
   141     *  used to determine the order in which the interrupts are serviced. In this
   142     *  case, lower interrupts have a higher priority i.e. interrupt 0 will have
   143     *  a higher priority than interrupt 31.
   144     *
   145     *  @a(Interrupt Masking Options)
   146     *  In this Hwi module, the {@link #maskSetting} instance configuration
   147     *  parameter is ignored.
   148     *  Effectively, only the {@link #MaskingOption_LOWER} is supported.
   149     *
   150     *  @a(More Hwi examples)
   151     *  Here's an example showing how to construct a Hwi at runtime:
   152     *  @p(code)
   153     *  *.c:
   154     *  #include <ti/sysbios/family/arm/pl192/Hwi.h>
   155     *
   156     *  Hwi_Struct hwiStruct;
   157     *
   158     *  Void myIsrIRQ(UArg arg)
   159     *  {
   160     *      ...
   161     *  }
   162     *
   163     *  Void main(Void)
   164     *  {
   165     *      Hwi_Params hwiParams;
   166     *
   167     *      Hwi_Params_init(&hwiParams);
   168     *      Hwi_construct(&hwiStruct, INT_NUM_IRQ, myIsrIRQ, &hwiParams, NULL);
   169     *      ...
   170     *      BIOS_start();
   171     *  }
   172     *  @p
   173     *
   174     *  @p(html)
   175     *  <h3> Calling Context </h3>
   176     *  <table border="1" cellpadding="3">
   177     *    <colgroup span="1"></colgroup> <colgroup span="5" align="center"></colgroup>
   178     *
   179     *    <tr><th> Function                 </th><th>  Hwi   </th><th>  Swi   </th><th>  Task  </th><th>  Main  </th><th>  Startup  </th></tr>
   180     *    <!--                                                                                                                 -->
   181     *    <tr><td> {@link #clearInterrupt}   </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
   182     *    <tr><td> {@link #create}           </td><td>   N    </td><td>   N    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   183     *    <tr><td> {@link #disable}          </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
   184     *    <tr><td> {@link #disableInterrupt} </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   185     *    <tr><td> {@link #disableIRQ}       </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
   186     *    <tr><td> {@link #enable}           </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td><td>   N    </td></tr>
   187     *    <tr><td> {@link #enableInterrupt}  </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   188     *    <tr><td> {@link #enableIRQ}        </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td><td>   N    </td></tr>
   189     *    <tr><td> {@link #getHandle}        </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   190     *    <tr><td> {@link #Params_init}      </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
   191     *    <tr><td> {@link #restore}          </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
   192     *    <tr><td> {@link #restoreInterrupt} </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
   193     *    <tr><td> {@link #restoreIRQ}       </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td></tr>
   194     *    <tr><td> {@link #construct}        </td><td>   N    </td><td>   N    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   195     *    <tr><td> {@link #delete}           </td><td>   N    </td><td>   N    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   196     *    <tr><td> {@link #destruct}         </td><td>   N    </td><td>   N    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   197     *    <tr><td> {@link #getHookContext}   </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   198     *    <tr><td> {@link #reconfig}         </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   199     *    <tr><td> {@link #setFunc}          </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   200     *    <tr><td> {@link #setHookContext}   </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   Y    </td><td>   N    </td></tr>
   201     *    <tr><td colspan="6"> Definitions: <br />
   202     *       <ul>
   203     *         <li> <b>Hwi</b>: API is callable from a Hwi thread. </li>
   204     *         <li> <b>Swi</b>: API is callable from a Swi thread. </li>
   205     *         <li> <b>Task</b>: API is callable from a Task thread. </li>
   206     *         <li> <b>Main</b>: API is callable during any of these phases: </li>
   207     *           <ul>
   208     *             <li> In your module startup after this module is started (e.g. Hwi_Module_startupDone() returns TRUE). </li>
   209     *             <li> During xdc.runtime.Startup.lastFxns. </li>
   210     *             <li> During main().</li>
   211     *             <li> During BIOS.startupFxns.</li>
   212     *           </ul>
   213     *         <li> <b>Startup</b>: API is callable during any of these phases:</li>
   214     *           <ul>
   215     *             <li> During xdc.runtime.Startup.firstFxns.</li>
   216     *             <li> In your module startup before this module is started (e.g. Hwi_Module_startupDone() returns FALSE).</li>
   217     *           </ul>
   218     *       </ul>
   219     *    </td></tr>
   220     *
   221     *
   222     *  </table>
   223     *  @p

   294        /*!
   295         *  ======== core0VectorTableAddress ========
   296         *  Determines the location of Core0's Interrupt Vector Table on a
   297         *  Dual-Core device. Default is device dependent.
   298         *
   299         *  On Dual-Core devices, both Cortex-R5 cores share a common reset
   300         *  vector table. In order to allow the 2 cores to register their own
   301         *  exception handlers, each core generates its own clone of the reset
   302         *  vector table and initializes it with its own exception handler
   303         *  addresses. The core specific vector tables are placed at fixed
   304         *  addresses so that the exception handler functions called by the
   305         *  common reset vector table known each core's vector table address
   306         *  and are able to reference it once they detect which core the
   307         *  application is currently running on.
   308         *
   309         *  The address of Core0's vector table is determined by this parameter.
   310         *
   311         *  Here are the default Core0 vector table addresses for all supported
   312         *  Dual-Core devices:
   313         *  @p(code)
   314         *   ----------------------------------------------------
   315         *  | Device name | Core0's default vector table address |
   316         *   ----------------------------------------------------
   317         *  | RM57D8xx    | 0x100                                |
   318         *   ----------------------------------------------------
   319         *  @p
   320         *
   321         *  @a(Note)
   322         *  If changing Core0's vector table address, it is not necessary to
   323         *  rebuild Core1's application as it does not need to know the location
   324         *  of Core0's vector table.
   325         */
   326        metaonly config Ptr core0VectorTableAddress;
   327    
   328        /*!
   329         *  ======== core1VectorTableAddress ========
   330         *  Determines the location of Core1's Interrupt Vector Table on a
   331         *  Dual-Core device. Default is device dependent.
   332         *
   333         *  On Dual-Core devices, both Cortex-R5 cores share a common reset
   334         *  vector table. In order to allow the 2 cores to register their own
   335         *  exception handlers, each core generates its own clone of the reset
   336         *  vector table and initializes it with its own exception handler
   337         *  addresses. The core specific vector tables are placed at fixed
   338         *  addresses so that the exception handler functions called by the
   339         *  common reset vector table known each core's vector table address
   340         *  and are able to reference it once they detect which core the
   341         *  application is currently running on.
   342         *
   343         *  Core0's vector table is always placed at 0x100 while the address of
   344         *  Core1's vector table is determined by this parameter.
   345         *
   346         *  Here are the default Core1 vector table addresses for all supported
   347         *  Dual-Core devices:
   348         *  @p(code)
   349         *   ----------------------------------------------------
   350         *  | Device name | Core1's default vector table address |
   351         *   ----------------------------------------------------
   352         *  | RM57D8xx    | 0x200000                             |
   353         *   ----------------------------------------------------
   354         *  @p
   355         *
   356         *  @a(Note)
   357         *  If changing Core1's vector table address, it is important to rebuild
   358         *  Core0's application with the same change as Core0 owns the common
   359         *  reset vector table and the common exception handler functions need
   360         *  to know Core1's vector table address so they can determine the
   361         *  address of the handler function they need to jump to.
   362         */
   363        metaonly config Ptr core1VectorTableAddress;

   371         *  vector table which points to a set of common vector handler

   529        /*!
   530         *  ======== disable ========
   531         *  Globally disable interrupts.
   532         *
   533         *  Hwi_disable globally disables hardware interrupts and returns an
   534         *  opaque key indicating whether interrupts were globally enabled or
   535         *  disabled on entry to Hwi_disable().
   536         *  The actual value of the key is target/device specific and is meant
   537         *  to be passed to Hwi_restore().
   538         *
   539         *  Call Hwi_disable before a portion of a function that needs
   540         *  to run without interruption. When critical processing is complete, call
   541         *  Hwi_restore or Hwi_enable to reenable hardware interrupts.
   542         *
   543         *  Servicing of interrupts that occur while interrupts are disabled is
   544         *  postponed until interrupts are reenabled. However, if the same type
   545         *  of interrupt occurs several times while interrupts are disabled,
   546         *  the interrupt's function is executed only once when interrupts are
   547         *  reenabled.
   548         *
   549         *  A context switch can occur when calling Hwi_enable or Hwi_restore if
   550         *  an enabled interrupt occurred while interrupts are disabled.
   551         *
   552         *  Hwi_disable may be called from main(). However, since Hwi interrupts
   553         *  are already disabled in main(), such a call has no effect.
   554         *
   555         *  @a(constraints)
   556         *  If a Task switching API such as
   557         *  {@link ti.sysbios.knl.Semaphore#pend Semaphore_pend()},
   558         *  {@link ti.sysbios.knl.Semaphore#post Semaphore_post()},
   559         *  {@link ti.sysbios.knl.Task#sleep Task_sleep()}, or
   560         *  {@link ti.sysbios.knl.Task#yield Task_yield()}
   561         *  is invoked which results in a context switch while
   562         *  interrupts are disabled, an embedded call to
   563         *  {@link #enable Hwi_enable} occurs
   564         *  on the way to the new thread context which unconditionally re-enables
   565         *  interrupts. Interrupts will remain enabled until a subsequent
   566         *  {@link #disable Hwi_disable}
   567         *  invocation.
   568         *
   569         *  Swis always run with interrupts enabled.
   570         *  See {@link ti.sysbios.knl.Swi#post Swi_post()} for a discussion Swis and
   571         *  interrupts.
   572         *
   573         *  @b(returns)     opaque key for use by Hwi_restore()
   574         */
   575        @Macro
   576        override UInt disable();

   673         *  The Vectored Interrupt Controllers (VIC) on Keystone3 are
   674         *  connected in a daisy-chain configuration (i.e. VIC1's output
   675         *  line is connected to VIC0's input line and so on) with each VIC

   686         *  The default value of -1 is used as a flag to indicate
   687         *  the lowest (logical) device-specific priority value.
